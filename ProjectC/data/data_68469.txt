	function get_play_index(idx, sec_id){
		var loc = loc_map[idx];
		for (var i=0;i<loc.length;i++){
			if (loc[i].sec == sec_id){
				return loc[i].index;
			}
		}
		return -1;
	}

	function movePoint(action_id, second) {
		var idx = getMapIndex(action_id);
		var play_index = get_play_index(idx, second);
		if (play_index != -1){

		}
	};

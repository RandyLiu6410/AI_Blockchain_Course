		var prev_sec = 0;

			"timeupdate",
			function(event){
				var player_id = this.id.split("_");
				$("#map_" + player_id[1]).show();
				$("#button_" + player_id[1]).hide();

				var sec = parseInt(this.currentTime);
				if (prev_sec != sec){
					prev_sec = sec;
									}
			}

		$("[id^=player]").on(
			"ended",
			function(event){

				$("#map_" + player_id[1]).hide();
				$("#button_" + player_id[1]).show();
				clearMarker(player_id[1]);
			}
		);

			"pause",
			function(event){
				var player_id = this.id.split("_");

			}

		$("[id^=player]").on(
			"ended",
			function(event){
				var player_id = this.id.split("_");
				$("#map_" + player_id[1]).hide();
				$("#button_" + player_id[1]).show();
				clearMarker(player_id[1]);
			}
		);

		$("[id^=player]").on(
			"pause",
			function(event){
				var player_id = this.id.split("_");
				$("#map_" + player_id[1]).hide();
				$("#button_" + player_id[1]).show();
			}
		);
	});
